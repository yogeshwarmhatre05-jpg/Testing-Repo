/* RGB THEME */ const themes = [ { bg:"#0F172A", border:"#22C55E", glow:"rgba(34,197,94,0.5)", text:"#22C55E" }, { bg:"#1E1B4B", border:"#A855F7", glow:"rgba(168,85,247,0.5)", text:"#A855F7" }, { bg:"#111827", border:"#3B82F6", glow:"rgba(59,130,246,0.5)", text:"#3B82F6" }, { bg:"#2A0F0F", border:"#FF4D4D", glow:"rgba(255,77,77,0.5)", text:"#FF4D4D" }, { bg:"#1A1A00", border:"#FFCC00", glow:"rgba(255,204,0,0.5)", text:"#FFCC00" } ]; let index = 0; function changeTheme(){ const theme = themes[index]; document.body.style.background = theme.bg; document.querySelector(".mainBox").style.borderColor = theme.border; document.querySelector(".mainBox").style.boxShadow = 0 0 25px ${theme.glow}; document.querySelector(".player").style.borderColor = theme.border; document.querySelector(".calendar").style.borderColor = theme.border; document.querySelector(".section").style.borderColor = theme.border; document.querySelector("#game").style.borderColor = theme.border; document.querySelector(".pixelBorder").style.background = linear-gradient( 90deg, ${theme.border}, white, ${theme.border} ); document.querySelector(".pixelBorder").style.backgroundSize = "300% 300%"; document.querySelector(".pixelBorder").style.animation = "discoRGB 6s ease-in-out infinite"; document.querySelectorAll(".themeText").forEach(el=>{ el.style.color = theme.text; el.style.textShadow = `0 0 12px ${theme.glow}`; }); document.getElementById("pixelText").style.color = theme.text; document.getElementById("pixelText").style.textShadow = 0 0 12px ${theme.glow}; index++; if(index >= themes.length){ index = 0; } } setInterval(changeTheme,6000); changeTheme();
